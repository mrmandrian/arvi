// KALIBRASI & TIMING RENDER BARU (Viewport-Oriented) activeModel.addEventListener('model-loaded', () => { // Buffer waktu 250ms untuk GPU setTimeout(() => { const obj = activeModel.object3D; if (obj) { obj.updateMatrixWorld(true); const box = new THREE.Box3().setFromObject(obj); const size = new THREE.Vector3(); box.getSize(size); // Mencari sisi paling panjang dari model (Bisa Panjang, Lebar, atau Tingginya) const maxDim = Math.max(size.x, size.y, size.z); // MAGIC VIEWPORT RATIO: // 3.5 artinya sisi terpanjang model akan dibuat 3.5x lipat lebih besar dari ukuran kertas QR. // Pada jarak pandang normal, ini akan membuat model pas menyentuh batas kiri & kanan layar HP. const viewportTarget = 3.5; const autoScale = viewportTarget / maxDim; scale = autoScale; // Simpan sebagai baseScale agar kalau di-Double Tap (Reset), baliknya ke ukuran layar ini models[currentModelId].baseScale = autoScale; controller.setAttribute('scale', `${scale} ${scale} ${scale}`); } // Loader hilang, model tampil megah di layar loaderContainer.style.display = 'none'; }, 250); });
